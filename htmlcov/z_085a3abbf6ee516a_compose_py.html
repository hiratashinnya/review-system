1"""P2 基準合成（MVP は org 最小）。design/02・schema。 

2 

3MVP：org の同 doc_type 兄弟ファイルを union（同 id は先勝ち）し、PF へ渡す CriteriaPack 

4（メタ抜き）と内部 MetaIndex（id→メタ）に分ける。enabled=False は pack から除外。 

5方向ゲート・本文矛盾・継承マージ（team/project）は MVP 外（PR8 印・post-MVP）。 

6""" 

7from __future__ import annotations 

8 

9import hashlib 

10 

11from ..domain.criteria import ComposedRule, CriteriaPack, PackedRule, MetaIndex, RuleMeta

15def build_pack_and_meta(rules: tuple[ComposedRule, ...]) -> tuple[CriteriaPack, MetaIndex]: 

16 """合成済みルール列 → (観点パック, メタ表)。同 id は先勝ち union、disabled は pack から除外。""" 

17 packed: list[PackedRule] = [] 

18 meta: dict[RuleId, RuleMeta] = {} 

19 seen: set[RuleId] = set() 

20 for r in rules: 

21 if r.rule_id in seen: # 兄弟 union：同 id は先勝ち（矛盾検出は post-MVP） 

22 continue 

23 seen.add(r.rule_id) 

24 meta[r.rule_id] = r.meta 

25 if r.meta.enabled: # disabled は PF に渡さない 

26 packed.append(PackedRule(r.rule_id, r.title, r.guidance)) 

27 return CriteriaPack(tuple(packed)), MetaIndex(meta)

29 

30def content_hash(rules: tuple[ComposedRule, ...]) -> ContentHash: 

31 """合成結果の決定的ハッシュ（S6 版スタンプ・再現性）。id＋メタ＋本文要約を採る。""" 

32 lines = sorted( 

33 f"{r.rule_id.value}|{r.meta.determinism.value}|{int(r.meta.severity)}|" 

34 f"{r.meta.override.value}|{r.meta.enabled}|{r.guidance.summary}" 

35 for r in rules 

36 ) 

37 digest = hashlib.sha256("\n".join(lines).encode("utf-8")).hexdigest() 

38 return ContentHash(digest)
