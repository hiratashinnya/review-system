1"""P5 適用（🤖 自動修正を内部 git に finding 単位コミット）。design/02・S4。 

2 

3🤖 の確定 fix を workspace に書き込み finding 単位でコミット。1件でも失敗したら実行ぶんを 

4基準点へ戻す（all-or-nothing＝S4：半端を残さない）。MVP は SuggestedFix.diff を「新しい 

5ファイル内容」として適用する（unified diff 適用は post-MVP）。 

6""" 

7from __future__ import annotations 

8 

9from ..domain.ids import ExecutionId 

10from ..domain.report import AppliedCommit 

11from ..domain.review import TriagedFinding, finding_key 

12from ..domain.result import StageOutcome, FailureStage, ok, fail 

13 

14 

15def apply_auto( 

16 exec_id: ExecutionId, 

17 auto: tuple[TriagedFinding, ...], 

18 targets: dict[str, str], 

19 workspace, 

20 now: str, 

21) -> StageOutcome[tuple[AppliedCommit, ...]]: 

22 """🤖 区分を内部 git に適用。失敗時は基準点へ全戻し（書込ゼロ）。 

23 

24 open() も含めて try 内で受ける（open 失敗も fail-close）。rollback は best-effort で、 

25 二次例外が原因例外を潰さないようにする（S4 の「書込ゼロ」を例外漏れ無しで保つ）。 

26 """ 

27 commits: list[AppliedCommit] = [] 

28 try: 

29 workspace.open(exec_id, targets) 

30 for tf in auto: 

31 fix = tf.finding.suggested_fix 

32 if fix is None: # 修正案が無い 🤖 はスキップ（適用対象外） 

33 continue 

34 rel = str(tf.finding.location.file) 

35 key = finding_key(tf.finding) 

36 ref = workspace.commit_fix(exec_id, key, rel, fix.diff) 

37 commits.append(AppliedCommit(exec_id, key, ref, now)) 

38 return ok(tuple(commits)) 

39 except Exception as e: # S4：途中失敗→実行ぶんを巻き戻す 

40 try: 

41 workspace.rollback_execution(exec_id) 

42 except Exception: # 二次例外で原因を潰さない 

43 pass 

44 return fail(FailureStage.APPLY, f"自動適用に失敗: {e}", None, 

45 "実行ぶんは基準点へ戻したので（書込ゼロ）、原因を確認のうえ手動で修正・適用してください。")
