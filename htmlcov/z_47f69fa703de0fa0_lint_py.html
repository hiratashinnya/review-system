2 

3パーサが読めた dict を、値レベルで検証する：必須キー・override/severity/determinism の語彙・ 

4id 一意性・version の MAJOR 対応・extends 先の存在。失敗は O-14 同形式（reason＋subject）。 

5"""

10from typing import Callable

12# version は MAJOR.MINOR(.PATCH…) のドット形式のみ（各成分が数字）＝M2 

13_VERSION = re.compile(r"(\d+)(?:\.\d+)+") 

14 

15# 対応する MAJOR（コメントでなく定数＝reviewer version / lint / 版スタンプの単一ソース・DD7） 

16SUPPORTED_CRITERIA_MAJOR: frozenset[int] = frozenset({1}) 

17SUPPORTED_POLICY_MAJOR: frozenset[int] = frozenset({1}) 

18 

19_OVERRIDE = {"locked", "tighten-only", "open"} 

20_SEVERITY = {"error", "warning", "info"} 

21_DETERMINISM = {"deterministic", "tradeoff", "judgment"} 

22_REQUIRED = ("doc_type", "scope", "rules")

32class CriteriaLintResult:

40def major_of(version: object) -> int | None: 

41 """`"MAJOR.MINOR"` 文字列から MAJOR(int) を取り出す（DD7）。 

42 

43 MAJOR.MINOR の**ドット形式（各成分が数字）以外は受理しない**（M2）。 

44 例：`"1.0"`→1、`"1"`/`"1."`/`"x.y"`→None（lint で fail-close）。 

45 """ 

46 if not isinstance(version, str): 

47 return None 

48 m = _VERSION.fullmatch(version.strip()) 

49 return int(m.group(1)) if m else None

52def lint_criteria(parsed: dict, *, exists: Callable[[str], bool]) -> CriteriaLintResult: 

53 """基準フロントマター dict を検証し、CriteriaLintResult を返す。""" 

54 errors: list[LintError] = [] 

55 

56 for key in _REQUIRED: 

57 if key not in parsed: 

58 errors.append(LintError(f"必須キーが無い: {key}")) 

59 

60 version = parsed.get("version") 

61 if version is not None: 

62 major = major_of(version) 

63 if major is None: 

64 errors.append(LintError(f"version は \"MAJOR.MINOR\" 文字列: {version!r}")) 

65 elif major not in SUPPORTED_CRITERIA_MAJOR: 

66 errors.append(LintError( 

67 f"未対応の基準 MAJOR={major}（対応={sorted(SUPPORTED_CRITERIA_MAJOR)}）", str(version))) 

68 

69 extends = parsed.get("extends") 

70 if extends not in (None, ""): 

71 if not exists(str(extends)): 

72 errors.append(LintError("extends 先が存在しない（継承リンク切れ）", str(extends))) 

73 

74 rules = parsed.get("rules") 

75 if rules is not None and not isinstance(rules, list): 

76 errors.append(LintError("rules はブロック列であるべき")) 

77 rules = [] 

78 

79 seen: set[str] = set() 

80 for idx, rule in enumerate(rules or []): 

81 if not isinstance(rule, dict): 

82 errors.append(LintError(f"rules[{idx}] はマッピングであるべき")) 

83 continue 

84 rid = rule.get("id") 

85 if not rid: 

86 errors.append(LintError(f"rules[{idx}] に id が無い")) 

87 else: 

88 if rid in seen: 

89 errors.append(LintError(f"rule id が重複: {rid}", str(rid))) 

90 seen.add(rid) 

91 _check_enum(rule, "override", _OVERRIDE, errors, rid) 

92 _check_enum(rule, "severity", _SEVERITY, errors, rid) 

93 _check_enum(rule, "determinism", _DETERMINISM, errors, rid) 

94 

95 return CriteriaLintResult(tuple(errors))

99 if key in rule and rule[key] not in allowed:

101 f"{key} は {sorted(allowed)} のいずれか: {rule[key]!r}", str(rid)))
