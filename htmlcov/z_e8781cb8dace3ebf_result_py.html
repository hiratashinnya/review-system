1"""失敗を型で表す（Result / StageOutcome）— S3 fail-close。design/01 §5。 

2 

3各段は例外を投げず、成功か FailureNotice を値で返す。呼び出し側は match で 

4Success / Failure を漏れなく分岐し、上流 Failure では副作用を走らせない。 

5""" 

6from __future__ import annotations 

7 

8from dataclasses import dataclass 

9from enum import Enum 

10from pathlib import Path 

11from typing import Generic, TypeVar 

12 

13T = TypeVar("T") 

14 

15 

16class FailureStage(Enum): 

17 INTAKE = "intake" 

18 COMPOSE = "compose" 

19 EVALUATE = "evaluate" 

20 VALIDATE = "validate" 

21 APPLY = "apply" 

22 LINT = "lint" # S5 事前 lint も O-14 形式で返す（DD12） 

23 

24 

25@dataclass(frozen=True, slots=True) 

26class FailureNotice: # O-14 異常系エラー通知 

27 stage: FailureStage 

28 reason: str 

29 subject: Path | None # どのファイル/対象（DD13: pathlib.Path） 

30 next_action: str # 実行可能な次手 

31 

32 

33@dataclass(frozen=True, slots=True) 

34class Success(Generic[T]): 

35 value: T 

36 

37 

38@dataclass(frozen=True, slots=True) 

39class Failure: 

40 notice: FailureNotice 

41 

42 

43# 各段の戻り値。空文書 no-op は Success(EmptyReport) で表す。 

44StageOutcome = Success[T] | Failure 

45 

46 

47def ok(value: T) -> Success[T]: 

48 """成功を短く構築（呼び出し側の意図を明確に）。""" 

49 return Success(value) 

50 

51 

52def fail(stage: FailureStage, reason: str, subject: Path | None, next_action: str) -> Failure: 

53 """fail-close を短く構築。O-14 の素材をそのまま渡す。""" 

54 return Failure(FailureNotice(stage=stage, reason=reason, subject=subject, next_action=next_action))
