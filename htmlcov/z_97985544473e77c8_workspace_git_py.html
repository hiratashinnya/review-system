1"""DS3 finding-commit ワークスペース（内部 git）。design/05・DD5・S4。 

2 

3実行ごとに対象を作業コピーへ展開し `git` を subprocess 駆動。finding 単位コミット（Q3）、 

4実行内失敗は基準点へ reset（all-or-nothing＝S4）、revert で個別/実行ぶんを戻す（O-6）。 

5git バイナリ前提（DD5）。Python 外部依存なし（stdlib subprocess）。 

6""" 

7from __future__ import annotations 

8 

9import re 

10import shutil 

11import subprocess 

12from pathlib import Path 

13 

14from ..domain.ids import ExecutionId 

15 

16_SAFE = re.compile(r"[^A-Za-z0-9._-]") 

17# 一時 git のコミット author＋署名無効（環境の git 設定に依存しない） 

18_IDENT = [ 

19 "-c", "user.email=reviewer@local", "-c", "user.name=reviewer", 

20 "-c", "commit.gpgsign=false", 

21] 

22 

23 

24def _safe_join(wd: Path, rel: str) -> Path: 

25 """`rel` を `wd` 配下の安全なパスに解決。絶対パス・`..` 脱出は ValueError（書込前に弾く）。""" 

26 target = (wd / rel).resolve() 

27 try: 

28 target.relative_to(wd.resolve()) 

29 except ValueError: 

30 raise ValueError(f"ワークスペース外への書込は不可: {rel!r}") 

31 return target 

32 

33 

34class GitWorkspaceRepository: 

35 def __init__(self, root: Path) -> None: 

36 self.root = root 

37 self._baseline: dict[str, str] = {} 

38 self._targets: dict[str, set[Path]] = {} # exec → 展開した評価対象集合（書込許可セット） 

39 

40 def workdir(self, exec_id: ExecutionId) -> Path: 

41 return self.root / _SAFE.sub("-", exec_id.value) 

42 

43 def open(self, exec_id: ExecutionId, files: dict[str, str]) -> None: 

44 """対象を作業コピーへ**作り直して**展開し git init、基準点コミットを打つ（残骸排除）。""" 

45 wd = self.workdir(exec_id) 

46 if wd.exists(): # N1：過去の残骸が混入しないよう作り直す 

47 shutil.rmtree(wd) 

48 wd.mkdir(parents=True, exist_ok=True) 

49 allowed: set[Path] = set() 

50 for rel, content in files.items(): 

51 p = _safe_join(wd, rel) # 脱出パスは拒否（防御の二重化） 

52 p.parent.mkdir(parents=True, exist_ok=True) 

53 p.write_text(content, encoding="utf-8") 

54 allowed.add(p) 

55 self._targets[exec_id.value] = allowed 

56 self._run(wd, "init", "-q") 

57 self._run(wd, "add", "-A") 

58 self._run(wd, *_IDENT, "commit", "-q", "-m", "baseline") 

59 self._baseline[exec_id.value] = self._head(wd) 

60 

61 def commit_fix(self, exec_id: ExecutionId, finding_id: str, rel_path: str, new_content: str) -> str: 

62 """1 finding 分の修正を書き込み、finding 単位でコミットして commit ref を返す。""" 

63 wd = self.workdir(exec_id) 

64 target = _safe_join(wd, rel_path) # LLM 由来の脱出パスをここで弾く（[10]） 

65 if target not in self._targets.get(exec_id.value, set()): # N1：評価対象集合のみ書込 

66 raise ValueError(f"評価対象集合に無いパスへの書込は不可: {rel_path!r}") 

67 target.write_text(new_content, encoding="utf-8") 

68 self._run(wd, "add", str(target.relative_to(wd.resolve()))) 

69 self._run(wd, *_IDENT, "commit", "-q", "-m", f"{exec_id.value} {finding_id}") 

70 return self._head(wd) 

71 

72 def rollback_execution(self, exec_id: ExecutionId) -> None: 

73 """実行内のいずれかが失敗 → 基準点へ全戻し（書込ゼロ＝S4 all-or-nothing）。""" 

74 baseline = self._baseline.get(exec_id.value) 

75 if baseline is None: # 基準点が無い（open 失敗等）＝戻すものが無い 

76 return 

77 self._run(self.workdir(exec_id), "reset", "--hard", "-q", baseline) 

78 

79 def revert(self, exec_id: ExecutionId, commit_ref: str) -> None: 

80 """指定コミットを revert（個別 finding／実行ぶんを戻す＝O-6）。""" 

81 wd = self.workdir(exec_id) 

82 self._run(wd, *_IDENT, "revert", "--no-edit", commit_ref) 

83 

84 def _head(self, wd: Path) -> str: 

85 return self._run(wd, "rev-parse", "HEAD").stdout.strip() 

86 

87 def _run(self, wd: Path, *args: str) -> subprocess.CompletedProcess: 

88 return subprocess.run( 

89 ["git", *args], cwd=wd, check=True, text=True, capture_output=True, 

90 )
