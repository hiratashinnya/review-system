87def _cmd_review(args) -> int: 

88 if not args.type_: 

89 print("error: --type が必要（MVP は AI 型推定なし）", file=sys.stderr) 

90 return EXIT_BADREQ 

91 request = ReviewRequest( 

92 targets=tuple(_read(p) for p in args.paths), 

93 references=tuple(_read(p) for p in args.ref), 

94 type_override=DocumentType(args.type_), 

95 scope=Scope.org(), 

96 ) 

97 adapter = (FilePlatformAdapter(Path(args.findings)) if args.findings 

98 else FakePlatformAdapter(RawReviewResponse((), ()))) 

99 platform = GuardingPlatform(adapter) # PF 境界のガード（例外→fail-close・M1） 

100 deps = Deps( 

101 platform=platform, 

102 load_criteria=lambda dt, sc: discover_criteria(Path(args.criteria), dt, sc), 

103 load_policy=lambda sc: load_policy_file(Path(args.policy)), 

104 now=_now, 

105 ) 

106 match run_review(request, deps): 

107 case Failure(notice): 

108 print(f"O-14 [{notice.stage.value}] {notice.reason} -> {notice.next_action}", 

109 file=sys.stderr) 

110 return EXIT_FAILCLOSE 

111 case Success(report): 

112 pass 

113 # 🤖 自動適用（finding 単位コミット・S4）。失敗時は書込ゼロで O-14 

114 if report.auto: 

115 targets = {str(sf.path): sf.content for sf in request.targets} 

116 out = apply_auto(report.stamp.execution_id, report.auto, targets, 

117 GitWorkspaceRepository(WORKSPACE), deps.now()) 

118 match out: 

119 case Failure(notice): 

120 print(f"O-14 [apply] {notice.reason} -> {notice.next_action}", file=sys.stderr) 

121 return EXIT_FAILCLOSE 

122 case Success(commits): 

123 _save_commits(report.stamp.execution_id.value, [c.commit_ref for c in commits]) 

124 Path(args.out).write_text(render_html(report), encoding="utf-8") 

125 print(args.out) 

126 return EXIT_OK

129def _cmd_revert(args) -> int: 

130 rid = _review_id_of(Path(args.report)) 

131 if rid is None: 

132 print("error: レポートから review_id を読めない", file=sys.stderr) 

133 return EXIT_BADREQ 

134 try: # #12 T3/T5：壊れた state ファイルもクラッシュさせず fail-close 

135 refs = _load_commits(rid) 

136 except (json.JSONDecodeError, UnicodeDecodeError, OSError) as e: 

137 print(f"O-14 [apply] commits 状態ファイルを読めない: {e} " 

138 f"-> .review-state の当該 .commits.json を確認のこと", file=sys.stderr) 

139 return EXIT_FAILCLOSE 

140 if not isinstance(refs, list) or not all(isinstance(r, str) for r in refs): # #12 T4 

141 print(f"O-14 [apply] commits 状態ファイルの形式が不正: {refs!r} " 

142 f"-> .review-state の当該 .commits.json を確認のこと", file=sys.stderr) 

143 return EXIT_FAILCLOSE 

144 if not refs: 

145 print("revert 対象なし", file=sys.stderr) 

146 return EXIT_BADREQ 

147 repo = GitWorkspaceRepository(WORKSPACE) 

148 exec_id = ExecutionId(rid) 

149 for ref in reversed(refs): # 新しい順に戻す 

150 try: # #10：git 失敗（衝突/不正 ref/workdir 破損）を fail-close 

151 repo.revert(exec_id, ref) 

152 except (subprocess.CalledProcessError, OSError) as e: 

153 detail = getattr(e, "stderr", None) or getattr(e, "stdout", None) or e # #12 T2：git の実エラーを O-14 に 

154 print(f"O-14 [apply] revert に失敗（{ref}）: {str(detail).strip()} " 

155 f"-> 一部のみ revert された可能性。ワークスペースを手動確認のこと", 

156 file=sys.stderr) 

157 return EXIT_FAILCLOSE 

158 print(f"reverted {len(refs)} commit(s) for {rid}") 

159 return EXIT_OK

162def _cmd_feedback(args) -> int: 

163 report = Path(args.report) 

164 rid = _review_id_of(report) 

165 if rid is None: 

166 print("error: レポートから review_id を読めない", file=sys.stderr) 

167 return EXIT_BADREQ 

168 fb_path = report.parent / f"{rid}.feedback.json" 

169 if not fb_path.exists(): 

170 print(f"error: {fb_path.name} が無い（HTML で書き出して同ディレクトリに置く）", file=sys.stderr) 

171 return EXIT_BADREQ 

172 try: # #6：壊れた JSON で落ちない 

173 data = json.loads(fb_path.read_text(encoding="utf-8")) 

174 except (json.JSONDecodeError, OSError) as e: 

175 print(f"error: feedback.json を読めない: {e}", file=sys.stderr) 

176 return EXIT_BADREQ 

177 if not isinstance(data, dict) or data.get("review_id") != rid: # #6：別 review 誤取込を防ぐ 

178 print("error: feedback.json の review_id がレポートと一致しない", file=sys.stderr) 

179 return EXIT_BADREQ 

180 try: # #6/#7：item のキー欠落は BADREQ 

181 n = append_feedback(STATE / "feedback.jsonl", rid, data.get("feedback", [])) 

182 except (ValueError, TypeError) as e: 

183 print(f"error: feedback の内容が不正: {e}", file=sys.stderr) 

184 return EXIT_BADREQ 

185 print(f"recorded {n} feedback item(s)") 

186 return EXIT_OK

189def _review_id_of(report: Path) -> str | None: 

190 m = _RID.search(report.read_text(encoding="utf-8")) 

191 return m.group(1) if m else None

194def _commits_path(rid: str) -> Path: 

195 safe = re.sub(r"[^A-Za-z0-9._-]", "-", rid) 

196 return STATE / f"{safe}.commits.json"

199def _save_commits(rid: str, refs: list[str]) -> None: 

200 STATE.mkdir(parents=True, exist_ok=True) 

201 _commits_path(rid).write_text(json.dumps(refs), encoding="utf-8")

204def _load_commits(rid: str) -> list[str]: 

205 p = _commits_path(rid) 

206 return json.loads(p.read_text(encoding="utf-8")) if p.exists() else []
