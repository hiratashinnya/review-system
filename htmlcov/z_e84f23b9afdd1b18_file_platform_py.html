1"""findings JSON を読む PlatformPort 実装（PF 駆動 MVP の実体）。design/04・DD8。

3Claude（LLM 役）が L1 出力スキーマで書いた findings.json を読み、ドメイン型へ。 

4schema 不適合は ❓未分類化せず、ここでは素直にパースだけ（検証は core/triage）。 

5""" 

6from __future__ import annotations 

7 

8import json 

9from pathlib import Path 

10 

11from ..domain.ids import RuleId, Location, LineRange 

12from ..domain.review import Finding, UnmatchedFinding, SuggestedFix 

13from ..ports.platform import PlatformCapabilities, RawReviewResponse

15 

16class FilePlatformAdapter: 

17 def __init__(self, findings_path: Path, model_id: str = "claude-code") -> None: 

18 self._path = findings_path 

19 self._model = model_id 

20 

21 def capabilities(self) -> PlatformCapabilities: 

22 return PlatformCapabilities(True, False, False, self._model, "claude-code") 

23 

24 def review(self, pack, targets, references) -> RawReviewResponse: 

25 data = json.loads(self._path.read_text(encoding="utf-8")) 

26 findings = tuple(_finding(d) for d in data.get("findings", [])) 

27 unmatched = tuple(_unmatched(d) for d in data.get("unmatched", [])) 

28 return RawReviewResponse(findings, unmatched) 

29 

30 

31def _location(d: dict) -> Location: 

32 loc = d["location"] 

33 lr = loc.get("line_range") 

34 return Location(Path(loc["file"]), LineRange(lr[0], lr[1]) if lr else None) 

35 

36 

37def _fix(d: dict | None) -> SuggestedFix | None: 

38 return SuggestedFix(d["description"], d["diff"]) if d else None 

39 

40 

41def _finding(d: dict) -> Finding: 

42 return Finding(RuleId(d["rule_id"]), _location(d), d.get("rationale", ""), 

43 d.get("quote"), _fix(d.get("suggested_fix"))) 

44 

45 

46def _unmatched(d: dict) -> UnmatchedFinding: 

47 return UnmatchedFinding(d.get("description", ""), _location(d), _fix(d.get("suggested_fix")))
